## TypeScript Interfaces

### Store Role Types

```typescript
// Store-level roles
type StoreRole = 'owner' | 'admin';

interface StoreUser {
  id: string;
  userId: string;
  storeId: string;
  role: StoreRole;
  permissions: StorePermission[];
  createdAt: Date;
}

type StorePermission =
  | 'store:manage'
  | 'services:manage'
  | 'orders:manage'
  | 'finance:view'
  | 'finance:manage'
  | 'team:view'
  | 'team:create'
  | 'team:delete'
  | 'settings:manage'
  | 'subscription:manage'
  | 'api:manage'
  | 'admin:manage';
```

### Team Role Types

```typescript
// Team-level roles
type TeamRole = 'lead' | 'assistant' | 'worker';

interface TeamMember {
  id: string;
  teamId: string;
  userId: string;
  role: TeamRole;
  permissions: TeamPermission[];
  joinedAt: Date;
  invitedBy: string;
}

type TeamPermission =
  | 'dashboard:view'
  | 'members:view'
  | 'members:invite'
  | 'members:remove'
  | 'members:promote'
  | 'jobs:view'
  | 'jobs:create'
  | 'jobs:edit'
  | 'jobs:delete'
  | 'jobs:review'
  | 'payouts:view'
  | 'payouts:approve'
  | 'settings:manage';

// Assistant permission config (set by Team Lead)
interface AssistantConfig {
  canApprovePayout: boolean;
  canDeleteJob: boolean;
  canRemoveMember: boolean;
}
```
